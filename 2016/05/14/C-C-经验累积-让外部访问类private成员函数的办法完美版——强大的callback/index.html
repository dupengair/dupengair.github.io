title: 让外部访问类private成员函数的办法完美版——强大的callback
date: 2016-05-13 17:01:51
categories: C&C++经验累积
tag: C&C++经验累积
toc: true

  • 对于class定义的private成员，其外部一般是无法访问的，除非定义成友元的形式。今天在live库中看到一种采用回调函数的形式，通过class回调其外部的方法，来访问class内部private成员的方法，写了个实例如下：

    //类定义：

class Product
{
public:
    virtual void show(void) = 0;
    Product();
    virtual ~Product();

};

    class Product_B
    {
    public:
        typedef void (*FUNC_test_B)(int args);
    public:
       void caller_B(FUNC_test_B test_B)              //参数为函数指针，准备访问Product_A中的Private方法
        {
        test_B(111);
        }
       void test_B(int args);

    };


    class Product_A: public Product
    {
    public:
       void show(void);  

       Product_A(int var = 0);
       ~Product_A();
       int var;

       void test_call(Product_B* caller)     //回调Product_B中的方法，以自己的Private方法做参数，此时要求这个方法为static
        {
        caller->caller_B((Product_B::FUNC_test_B)& test_A);
        }
    private:
       static void test_A(int args)            //定义成静态方法
        {
        cout << "in Product_A : test_A" << endl;
        }

    };

    //客户端代码：

    int main()
    {  
        Product_B *pdt_B = new Product_B();

        Product_A AA;
        AA.test_call(pdt_B);

        delete pdt_B;
    }

+ 可以看到，在定义`Product_A`的`test_call()`方法时接受`Product_B`对象作为参数，回调`Product_B`定义的`caller_B()`方法，同时以`Product_A`的private方法`test_A`做`caller_B()`的参数，取得test_A方法的地址，进行类型转换后传给`caller_B()`方法。实现的思路是在`Product_A`中回调`Product_B`定义的`caller_B()`方法来访问自己的private成员。
+ 这里的取地址转换函数类型的写法，是先将`test_A()`方法定义成了static类型，所以可以直接对方法取地址，因为整个类里面只有一个方法存在，地址是唯一的，如果不是static，则要使用类成员指针的方法，找到当前P`roduct_A`对象的`test_A()`方法的地址再进行转换，会复杂一些。
  • 另外:下午试了一下如果不把成员函数定义成static，那么用类成员指针就要这么定义：
class Product_A: public Product
{
public:
    typedef void (Product_A::*FUNC_test_A)(int args);          //定义指向成员函数的指针
public:
    void show(void);       

//protected:
    Product_A(int var = 0);
    ~Product_A();
       int var;

   void test_call(Product_B* caller)
    {
    //caller->caller_B((Product_B::FUNC_test_B)& test_A);
    FUNC_test_A pFunc= &Product_A::test_A;                //取得成员函数的地址
    caller->caller_B((Product_B::FUNC_test_B)pFunc);        //进行类型转换
    }
private:
   //static void test_A(int args)
   void test_A(int args)                            //普通成员
    {
    cout << "in Product_A : test_A" << endl;
    }

};
