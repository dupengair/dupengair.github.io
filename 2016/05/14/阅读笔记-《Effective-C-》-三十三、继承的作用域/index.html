三十三、继承的作用域

文章目录

#一、函数与全局作用域

int x;
void fun()
{
    double x;    //local 变量隐藏全局变量
}

#二、继承的作用域

class Base {
private:
    int x;
public:
    virtual void mf1() = 0;
    virtual void mf2();
    void mf3();
};

class Derrived : public Base {
public:
    virtual void mf1();        //隐藏Base::mf1()
    void mf4();
};
  1. Derived class的作用域嵌套在base class内，隐藏base class的符号
  2. 唯一重要的是名称，类型并不重要

#三、重载函数的继承

class Base {
private:
    int x;
public:
    virtual void mf1() = 0;
    virtual void mf1(int);        //重载mf1() 
    virtual void mf2();
    void mf3();
    void mf3(double);        //重载mf3()
};
class Derived : public Base {
public:
    virtual void mf1();        //隐藏Base::mf1()、Base::mf1(int)
    void mf3();            //隐藏Base::mf3()、Base::mf3(double)
    void mf4();
};

Derived d;
int x;

d.mf1();    //调用 Derived::mf1()
d.mf1(x);    //error!
d.mf2();    //调用 Base::mf2()
d.mf3();    //调用 Derived::mf3()
d.mf3(x);    //error!
  1. 即使base class和derived class内函数参数类型不同作用域也适用
  2. 不论virtual与否都适用
  3. 如果通过多态的形式访问，则不会有问题（C++Primier 15.5.4）

#四、实质

  1. 是为了防止在程序库或应用框架（application framework）内建立新的derived class时附带地从疏远的base class继承重载函数。
  2. 实际上如果使用public继承又不继承重载函数，就违反继承的“is-a”关系。

#五、继承重载函数

class Base {
private:
    int x;
public:
    virtual void mf1() = 0;
    virtual void mf1(int);        //重载mf1() 
    virtual void mf2();
    void mf3();
    void mf3(double);        //重载mf3()
};
class Derived : public Base {
public:
    using Base::mf1;        //让Base class的符号在derived class中可见
    using Base::mf3;        //同上
    virtual void mf1();        
    void mf3();            
    void mf4();
};

Derived d;
int x;

d.mf1();    //调用 Derived::mf1()
d.mf1(x);    //调用 Base::mf1(int)
d.mf2();    //调用 Base::mf2()
d.mf3();    //调用 Derived::mf3()
d.mf3(x);    //调用 Base::mf3(double)
  1. 如果希望继承bass class的重载函数，又希望重新定义其中的一部分，可以引入using声明
  2. base class 内的public名称在public继承的derived class内也应该是public

#六、private继承

class Base {
public:
    virtual void mf1() = 0;
    virtual void mf1(int);        //重载mf1() 

};
class Derived : private Base {
public:
    virtual void mf1() { Base::mf1(); }     //forwarding function 自然成为inline，不用using的目的是只想mf1()        
};

Derived d;
int x;

d.mf1();    //调用 Derived::mf1()
d.mf1(x);    //error! Base::mf1(x)被隐藏了
  1. 在private继承中使用forwarding function（转交函数）使继承自Base class的符号可见
  2. forwarding function将继承得到的符号汇入derived class内
