关于SDP的接口API

sdp_message_t *eXosip_get_remote_sdp (int did);     //通过对话id获取远端(请求接收端)的sdp消息体
sdp_message_t *eXosip_get_local_sdp (int did);      //通过对话id获取本地(请求发送端)的sdp消息体
sdp_message_t *eXosip_get_previous_local_sdp (int did);   //获取之前发送INVITE请求的本地sdp消息体
sdp_message_t *eXosip_get_remote_sdp_from_tid (int tid);  //获取最近一个发送INVITE请求的远端sdp消息体
sdp_message_t *eXosip_get_local_sdp_from_tid (int tid);   //获取最近一个发送INVITE请求的本地sdp消息体
sdp_message_t *eXosip_get_sdp_info (osip_message_t * message);   //从sip消息中取出sdp消息体
sdp_connection_t *eXosip_get_audio_connection (sdp_message_t * sdp);   //从sdp消息中取出音频连接信息（m字段对应的会话描述）
sdp_media_t *eXosip_get_audio_media (sdp_message_t * sdp);   //从sdp消息中取出音频媒体信息
sdp_connection_t *eXosip_get_video_connection (sdp_message_t * sdp);   //从sdp消息中取出视频连接信息（m字段对应的会话描述）
sdp_media_t *eXosip_get_video_media (sdp_message_t * sdp);   //从sdp消息中取出视频媒体信息
sdp_connection_t *eXosip_get_connection (sdp_message_t * sdp, const char *media);  //根据媒体信息从sdp消息中取出媒体连接信息（m字段对应的会话描述）
sdp_media_t *eXosip_get_media (sdp_message_t * sdp, const char *media);//根据媒体信息从sdp消息中取出媒体信息（m字段对应的会话描述）
