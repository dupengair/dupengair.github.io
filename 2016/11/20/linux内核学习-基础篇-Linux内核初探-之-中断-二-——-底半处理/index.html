Linux内核初探 之 中断(二) —— 底半处理

一、下半部

  1. 中断的局限
    • 适合
      • 对时间敏感
      • 硬件相关
      • 保证不被其它任务打断
    • 局限
      • 异步执行，打断当前系统，越快越好
      • 最好情况屏蔽同级其它中断，最坏屏蔽当前处理器所有其它中断
      • 对硬件的操作有时限
      • 不在进程上下文运行，不能阻塞
  2. 执行
    • 中断返回后马上运行
    • 执行的关键在于运行时允许响应所有的中断
  3. 软中断
    • 32个，需要静态定义，编译期间静态注册
    • 可以在所有处理器上同时执行，要求必需多处理器上可重入
    • 性能要求高才使用，执行频率和连续性高，主要考虑多处理器并发执行(相同的软中断)
  4. tasklet
    • 动态创建与注册，对加锁的要求不高
    • 不同类型的tasklet可以在不同处理器上同时执行，类型相同不能同时执行
    • 性能和易用性间的平衡，更常用，通常应考虑

二、软中断

  1. 定义

    <linux/interrupt.h>
// 目前用到的软中断
enum
{
    HI_SOFTIRQ=0,
    ...

    NR_SOFTIRQS
};

/* softirq mask and active fields moved to irq_cpustat_t in
 * asm/hardirq.h to get better cache usage.  KAO
 */

struct softirq_action
{
    void    (*action)(struct softirq_action *);
};

<kernel/softirq.c>
static struct softirq_action softirq_vec[NR_SOFTIRQS] __cacheline_aligned_in_smp;

  1. 执行

    irq_exit()        [kernel/softirq.c] ->
invoke_softirq()                   ->
do_softirq()                       ->
__do_softirq()        

<kernel/softirq.c>
asmlinkage void __do_softirq(void)
{
    ...
    __local_bh_disable((unsigned long)__builtin_return_address(0), SOFTIRQ_OFFSET);
    lockdep_softirq_enter();

    /* Reset the pending bitmask before enabling irqs */
    set_softirq_pending(0);
    local_irq_enable();
    h = softirq_vec;

    do {
        if (pending & 1) {
            ...

            trace_softirq_entry(vec_nr);
            h->action(h);
            trace_softirq_exit(vec_nr);
            ...
        }
    } while (pending);

    local_irq_disable();
    ...

    // 发现已经执行过度内核线程重新触发了自己，唤醒软中断内核线程来处理
    pending = local_softirq_pending();
    if (pending && --max_restart)
        goto restart;

    if (pending)
        wakeup_softirqd();

    lockdep_softirq_exit();
    ...
}
    • 注册的软中断必须在标记后才会执行，称作触发，中断处理程序会在返回前标记它的软中断，使其稍后执行
    • 软中断之间不会相互抢占，只有中断可以抢占软中断，相同类型的软中断可以在其它处理器上同时运行

  2. 使用

    • 分配索引

      <linux/interrupt.h>
/* PLEASE, avoid to allocate new softirqs, if you need not _really_ high
   frequency threaded job scheduling. For almost all the purposes
   tasklets are more than enough. F.e. all serial device BHs et
   al. should be converted to tasklets, not to softirqs.
*/
// 目前用到的软中断
enum
{
    HI_SOFTIRQ=0,
    TIMER_SOFTIRQ,
    NET_TX_SOFTIRQ,
    NET_RX_SOFTIRQ,
    BLOCK_SOFTIRQ,
    BLOCK_IOPOLL_SOFTIRQ,
    TASKLET_SOFTIRQ,
    SCHED_SOFTIRQ,
    HRTIMER_SOFTIRQ,
    RCU_SOFTIRQ,    /* Preferable RCU should always be the last softirq */

    NR_SOFTIRQS
};
      • 根据希望赋予的优先级来决定加入位置，新项可能插入在BLOCK_SOFTIRQ和TASKLET_SOFTIRQ之间

    • 注册处理程序

      <kernel/softirq.c>
void open_softirq(int nr, void (*action)(struct softirq_action *))
{
    softirq_vec[nr].action = action;
}

<net/core/dev.c>
static int __init net_dev_init(void)
{
    ...
    open_softirq(NET_TX_SOFTIRQ, net_tx_action);
    open_softirq(NET_RX_SOFTIRQ, net_rx_action);
    ...
}            
subsys_initcall(net_dev_init);
      • 运行时通过open_softirq()注册中断处理程序
      • 中断处理程序执行时，允许响应中断，但不能休眠
      • 当前处理器上的软中断被禁止，但可以在别的处理器上被执行，所以要对共享数据严格执行锁保护
      • 锁影响软中断的并发执行，通常采用per-cpu data

  3. 触发

    <kernel/softirq.c>
/*
 * This function must run with irqs disabled!
 */

inline void raise_softirq_irqoff(unsigned int nr)
{
    __raise_softirq_irqoff(nr);

    /*
     * If we're in an interrupt or softirq, we're done
     * (this also catches softirq-disabled code). We will
     * actually run the softirq once we return from
     * the irq or softirq.
     *
     * Otherwise we wake up ksoftirqd to make sure we
     * schedule the softirq soon.
     */
    if (!in_interrupt())
        wakeup_softirqd();
}

// 设置软中断为挂起状态，下次调用`do_softirq()`时运行，触发前后要禁止和恢复中断
void raise_softirq(unsigned int nr)
{
    unsigned long flags;

    local_irq_save(flags);
    raise_softirq_irqoff(nr);
    local_irq_restore(flags);
}

void __raise_softirq_irqoff(unsigned int nr)
{
    trace_softirq_raise(nr);
    or_softirq_pending(1UL << nr);
}

三、tasklet

  1. 通常也是软中断
    • 由HI_SOFTIRQ和TASKLET_SOFTIRQ代表，前者更优先执行

  2. tasklet_struct结构

    <linux/interrupt.h>
/* Tasklets --- multithreaded analogue of BHs.

   Main feature differing them of generic softirqs: tasklet
   is running only on one CPU simultaneously.

   Main feature differing them of BHs: different tasklets
   may be run simultaneously on different CPUs.

   Properties:
   * If tasklet_schedule() is called, then tasklet is guaranteed
     to be executed on some cpu at least once after this.
   * If the tasklet is already scheduled, but its execution is still not
     started, it will be executed only once.
   * If this tasklet is already running on another CPU (or schedule is called
     from tasklet itself), it is rescheduled for later.
   * Tasklet is strictly serialized wrt itself, but not
     wrt another tasklets. If client needs some intertask synchronization,
     he makes it with spinlocks.
 */

struct tasklet_struct
{
    struct tasklet_struct *next;
    unsigned long state;
    atomic_t count;
    void (*func)(unsigned long);
    unsigned long data;
};

    • state成员

      <linux/interrupt.h>
enum
{
    // 已被调度，正准备投入运行
    TASKLET_STATE_SCHED,    /* Tasklet is scheduled for execution */
    // 正在运行
    TASKLET_STATE_RUN        /* Tasklet is running (SMP only) */
};

    • count

      • 为0时tasklet被激活
      • 不为0时tasklet被禁止

  3. 调度

    • tasklet_schedule()

      <linux/interrupt.h>
static inline void tasklet_schedule(struct tasklet_struct *t)
{
    // 状态时TASKLET_STATE_SCHED表明已经被调度过，不用再调度
    if (!test_and_set_bit(TASKLET_STATE_SCHED, &t->state))
        __tasklet_schedule(t);
}

<kernel/softirq.c>
void __tasklet_schedule(struct tasklet_struct *t)
{
    unsigned long flags;

    // 需要保持状态并禁止本地中断，保证状态不会乱
    local_irq_save(flags);
    t->next = NULL;
    // 要调度的tasklet需要加入每个处理器的vec链表
    *__this_cpu_read(tasklet_vec.tail) = t;
    __this_cpu_write(tasklet_vec.tail, &(t->next));
    // 中断唤醒后，下一次调用do_softirq()时会被执行
    raise_softirq_irqoff(TASKLET_SOFTIRQ);
    local_irq_restore(flags);
}

EXPORT_SYMBOL(__tasklet_schedule);

void __tasklet_hi_schedule(struct tasklet_struct *t)
{
    unsigned long flags;

    local_irq_save(flags);
    t->next = NULL;
    *__this_cpu_read(tasklet_hi_vec.tail) = t;
    __this_cpu_write(tasklet_hi_vec.tail,  &(t->next));
    raise_softirq_irqoff(HI_SOFTIRQ);
    local_irq_restore(flags);
}

  4. 执行

    • 注册软中断

      <kernel/softirq.c>
void __init softirq_init(void)
{
    int cpu;

    for_each_possible_cpu(cpu) {
        int i;

        per_cpu(tasklet_vec, cpu).tail =
            &per_cpu(tasklet_vec, cpu).head;
        per_cpu(tasklet_hi_vec, cpu).tail =
            &per_cpu(tasklet_hi_vec, cpu).head;
        for (i = 0; i < NR_SOFTIRQS; i++)
            INIT_LIST_HEAD(&per_cpu(softirq_work_list[i], cpu));
    }

    register_hotcpu_notifier(&remote_softirq_cpu_notifier);

    open_softirq(TASKLET_SOFTIRQ, tasklet_action);
    open_softirq(HI_SOFTIRQ, tasklet_hi_action);
}

    • 执行

      <kernel/softirq.c>
static void tasklet_action(struct softirq_action *a)
{
    struct tasklet_struct *list;

    // 因为tasklet总是被软中断调用，所以不用保存状态，软中断总是激活中断的，所以这里禁止中断后需要激活
    local_irq_disable();
    list = __this_cpu_read(tasklet_vec.head);
    __this_cpu_write(tasklet_vec.head, NULL);
    __this_cpu_write(tasklet_vec.tail, &__get_cpu_var(tasklet_vec).head);
    local_irq_enable();

    // 循环处理每一个tasklet
    while (list) {
        struct tasklet_struct *t = list;

        list = list->next;

        if (tasklet_trylock(t)) {
            if (!atomic_read(&t->count)) {
                if (!test_and_clear_bit(TASKLET_STATE_SCHED, &t->state))
                    BUG();
                t->func(t->data);
                tasklet_unlock(t);
                continue;
            }
            tasklet_unlock(t);
        }

        local_irq_disable();
        t->next = NULL;
        *__this_cpu_read(tasklet_vec.tail) = t;
        __this_cpu_write(tasklet_vec.tail, &(t->next));
        __raise_softirq_irqoff(TASKLET_SOFTIRQ);
        local_irq_enable();
    }
}
      • tasklet总在调度它的处理器上执行，可以更好的利用高速缓存

  5. 使用tasklet

    • 静态创建

      <linux/interrupt.h>
// 创建时处于激活状态，count值为0
#define DECLARE_TASKLET(name, func, data) \
struct tasklet_struct name = { NULL, 0, ATOMIC_INIT(0), func, data }

// 创建时处于禁止状态，count值为1
#define DECLARE_TASKLET_DISABLED(name, func, data) \
struct tasklet_struct name = { NULL, 0, ATOMIC_INIT(1), func, data }

    • 动态创建

      <sound/core/timer.c>
tasklet_init(&timer->task_queue, snd_timer_tasklet, (unsigned long)timer);

    • 编写tasklet处理程序

      <sound/core/timer.c>
/*
 * timer tasklet
 *
 */
static void snd_timer_tasklet(unsigned long arg)
{
    ...
}
      • 靠软中断实现，不允许睡眠，不能在其中使用信号量或者阻塞式函数
      • 需要注意共享数据用锁保护，因为允许中断

    • 调用tasklet

      <sound/core/timer.c>
/*
 * timer interrupt
 *
 * ticks_left is usually equal to timer->sticks.
 *
 */
void snd_timer_interrupt(struct snd_timer * timer, unsigned long ticks_left)
{
    ...
    if (use_tasklet)
        tasklet_schedule(&timer->task_queue);
}
      • tasklet被调度后，只要有机会就会尽早运行

    • 激活和屏蔽

      static inline void tasklet_enable(struct tasklet_struct *t);
static inline void tasklet_disable(struct tasklet_struct *t);

  6. ksoftirqd

    • 当内核中出现大量软中断的时候，内核进程就会辅助处理它们
    • 当软中断执行时可以重复触发自己得以再次执行，因而导致用户空间进程无法获得足够处理时间而处于饥饿状态
    • 大量软中断出现时，内核会唤醒一组ksoftirqd/n线程来处理负载，这些线程运行在最低优先级上(nice=19)，避免与重要任务竞争资源
    • 只要do_softirq()函数发现已经执行过度内核线程重新出发了自己，软中断内核线程就会被唤醒(见上面二.2软中断执行)
